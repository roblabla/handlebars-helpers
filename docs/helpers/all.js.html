var fs    = require('fs');
var path  = require('path');
var _     = require('lodash');

module.exports = {};

fs.readdirSync(path.join(__dirname)).forEach(function(file) {
  if (file != 'all.js')
    _.extend(module.exports.helpers.all, require('./'+ file));
});
